\chapter{预备知识与符号系统}
\section{预备知识}
本节将简要回顾与本文研究相关的椭圆型偏微分方程的基础理论与先验估计（Priori Estimates）。本文考察如下形式的椭圆型方程：
\begin{equation}\label{eq:general_elliptic}
  -\Delta u(x) + f(x, u) = g(x), \quad x \in \Omega,
\end{equation}
其中，$\Omega \subset \mathbb{R}^n$ 为具有足够光滑边界的有界开区域。当非线性项 $f(x, u)$ 关于未知函数 $u$ 呈线性依赖（例如 $f(x, u) = V(x)u(x)$）时，方程 \eqref{eq:general_elliptic} 退化为线性薛定谔方程或一般线性椭圆方程，此时 $V(x)$ 通常被称为势函数。反之，若 $f(x, u)$ 包含 $u$ 的高次幂或其他非线性结构，则该方程为非线性椭圆方程。

为了对解的适定性与正则性进行严密的理论分析，我们首先引入一般二阶线性椭圆算子 $L$ 的散度/非散度形式：
$$Lu = -\sum_{i,j=1}^n (a^{ij}(x) u_{x_i})_{x_j} + \sum_{i=1}^n b^i(x) u_{x_i} + c(x)u$$
我们称算子 $L$ 满足\textbf{一致椭圆性}条件，若：存在常数 $\theta > 0$，使得对于任意的 $x \in \Omega$ 和 $\xi \in \mathbb{R}^n$，均有 $\sum_{i,j=1}^n a^{ij}(x) \xi_i \xi_j \ge \theta |\xi|^2$。

基于上述定义，以下定理总结了椭圆方程解在不同函数空间中的正则性与先验估计结果：

\begin{lemma}[椭圆方程的内部 $L^2$ 估计]\cite{evans2010partial}
设 $\Omega$ 是 $\mathbb{R}^n$ 中的开集。假设算子 $L$ 满足一致椭圆性条件，且其系数满足 $a^{ij} \in C^1(\Omega)$，$b^i, c \in L^\infty(\Omega)$。
若 $u \in H^1(\Omega)$ 是方程 $Lu = f$ 在 $\Omega$ 中的弱解，且源项 $f \in L^2(\Omega)$，那么对于任意紧包含于 $\Omega$ 的子区域 $V$（即 $V \subset\subset \Omega$），均有 $u \in H^2(V)$，并且存在一个常数 $C$ 使得如下内部估计成立：
$$||u||_{H^2(V)} \le C \left( ||f||_{L^2(\Omega)} + ||u||_{L^2(\Omega)} \right)$$
\end{lemma}

\begin{lemma}[椭圆方程的全局 $L^2$ 估计]\cite{evans2010partial}
设 $\Omega \subset \mathbb{R}^n$ 是一个有界开集，且其边界 $\partial\Omega$ 是 $C^2$ 平滑的。假设算子 $L$ 满足一致椭圆性，且系数 $a^{ij} \in C^1(\bar{\Omega})$，$b^i, c \in L^\infty(\Omega)$。
若 $u \in H^1(\Omega)$ 是如下 Dirichlet 边值问题的弱解：
$$\begin{cases} 
Lu = f & \text{在 } \Omega \text{ 内} \\ 
u = g & \text{在 } \partial\Omega \text{ 上} 
\end{cases}$$
其中源项 $f \in L^2(\Omega)$，边界条件 $g \in H^{3/2}(\partial\Omega)$。那么解的全局正则性提升至 $u \in H^2(\Omega)$，并且存在一个仅依赖于 $n$, $\Omega$ 以及算子 $L$ 系数的常数 $C$，使得以下估计成立：
$$||u||_{H^2(\Omega)} \le C \left( ||f||_{L^2(\Omega)} + ||g||_{H^{3/2}(\partial\Omega)} + ||u||_{L^2(\Omega)} \right)$$
特别地，如果已知 $u$ 是该边值问题的唯一弱解（例如当 $c \ge 0$ 且无特征值共振时），则可以利用庞加莱不等式等技巧将右侧的 $||u||_{L^2(\Omega)}$ 项吸收。
\end{lemma}

尽管上述 $L^2$ 估计奠定了弱解存在的理论基础，但其仅局限于 Sobolev 空间内的弱导数范畴。在处理某些问题时，往往需要获取解的逐点连续性等更强的拓扑性质。De Giorgi-Nash 定理正是填补这一缺陷的关键，它在极宽泛的系数假设下赋予了弱解局部 Hölder 连续性：
\begin{lemma}[De Giorgi-Nash 估计]
考虑如下形式的齐次散度型方程：
$$-\sum_{i,j=1}^n \frac{\partial}{\partial x_i} \left( a_{ij}(x) \frac{\partial u}{\partial x_j} \right) = 0$$
假设区域边界满足外锥条件，系数矩阵 $A(x) = [a_{ij}(x)]$ 满足一致椭圆性条件，即存在常数 $\lambda, \Lambda > 0$ 使得对所有 $\xi \in \mathbb{R}^n$ 和几乎所有 $x \in \Omega$，有 $\lambda |\xi|^2 \leq \sum_{i,j=1}^n a_{ij}(x) \xi_i \xi_j \leq \Lambda |\xi|^2$。
De Giorgi-Nash 定理表明，即使系数 $a_{ij}$ 仅仅是有界可测的，方程的弱解 $u$ 在内部也是局部 Hölder 连续的。即存在指数 $\alpha \in (0,1)$，使得 $u \in C^{\alpha}_{\text{loc}}(\Omega)$。
\end{lemma}

进一步地，当区域边界与方程系数具备更好的平滑性时，解的正则性能够得到进一步提高。Calderón-Zygmund 理论将上述先验估计推广至更高阶的 Sobolev 空间和更一般的 $L^p$ 空间框架中：
\begin{lemma}[全局 $L^p$ 估计, Calderón-Zygmund 估计]\label{lem:lp}
对于一般的二阶散度型椭圆方程：
$$-\sum_{i,j=1}^n \frac{\partial}{\partial x_i} \left( a_{ij}(x) \frac{\partial u}{\partial x_j} \right) = f$$
设 $k$ 为非负整数，$1 < p < \infty$。若右端源项 $f \in W^{k,p}(\Omega)$，且系数 $a_{ij}$ 满足 $C^{k,1}(\bar{\Omega})$ 正则性，区域边界 $\partial \Omega$ 为 $C^{k+2}$ 级。那么，该方程的弱解 $u$ 属于 $W^{k+2,p}(\Omega)$，且满足如下 Calderón-Zygmund 型先验估计：
$$\|u\|_{W^{k+2,p}(\Omega)} \leq C \left( \|f\|_{W^{k,p}(\Omega)} + \|u\|_{L^p(\Omega)} \right)$$
\end{lemma}

在处理非齐次边界条件或进行全空间泛函分析时（例如在构建神经网络的全局逼近函数时），Sobolev 函数的延拓算子是不可或缺的分析工具：
\begin{lemma}[Calderón-Stein 延拓定理]\label{lem:extension}
设 $\Omega \subset \mathbb{R}^n$ 是一个有界 Lipschitz 区域。对于每一个非负整数 $k$ 和每一个 $1 \leq p \leq \infty$，存在一个有界的线性延拓算子 $E: W^{k,p}(\Omega) \to W^{k,p}(\mathbb{R}^n)$，使得对于所有的 $f \in W^{k,p}(\Omega)$，在 $\Omega$ 上几乎处处有 $(Ef)|_\Omega = f$，且存在一个仅依赖于 $k, p, n$ 和区域 $\Omega$ 几何性质的常数 $C$，使得：
$$\|Ef\|_{W^{k,p}(\mathbb{R}^n)} \leq C \|f\|_{W^{k,p}(\Omega)}$$
\end{lemma}

最后，在针对半线性及非线性问题的理论分析中，特别是当非线性势函数仅具备较低的局部可积性时，Brezis-Kato 定理在跨越临界指数、提升解的整体可积性方面起到了重要作用：
\begin{lemma}[Brezis-Kato 定理]\label{lem:Brezis}\cite{brezis1978remarks,willem2012minimax}
设 $\Omega$ 为 $\mathbb{R}^n$ ($n \ge 3$) 中的一个开集。假设势函数 $a(x)$ 满足 $a \in L_{\text{loc}}^{n/2}(\Omega)$。若 $u \in H_{\text{loc}}^1(\Omega)$ 是如下微分不等式的弱下解（Weak Subsolution）：
$$-\Delta u(x) \le a(x) u(x) \quad \text{in } \Omega$$
即对于任意的非负测试函数 $\varphi \in C_0^\infty(\Omega)$，均有积分不等式：
$$\int_{\Omega} \nabla u \cdot \nabla \varphi \, dx \le \int_{\Omega} a(x)u\varphi \, dx$$
则对于任意有限指数 $q \in [1, +\infty)$，均有 $u \in L_{\text{loc}}^q(\Omega)$。
\end{lemma}

\section{符号与定义}

\subsection{神经网络架构与参数}
本节将建立全连接深度神经网络的数学框架并引入相关记号。

\begin{definition}[全连接神经网络，Fully Connected Neural Network, FNN]\label{def:fnn}
我们将全连接深度神经网络表示为参数化函数 $v_{\phi}: \mathbb{R}^{d} \to \mathbb{R}$。该网络通过层级递归的方式定义如下：
\begin{equation}\label{eq:nn:fcnetwork}
\begin{aligned}
v_{0}(x) &= x, \\
v_{\ell}(x) &= \sigma_{\ell}\left(A_{\ell} v_{\ell-1} + b_{\ell}\right), \quad \ell=1, 2, \cdots, L-1, \\
v_{\phi}(x) &= v_{L}(x) = A_{L} v_{L-1} + b_{L},
\end{aligned}
\end{equation}
其中，$L$ 表示网络的层数（即深度）。对于第 $\ell$ 层，$\boldsymbol{A}_{\ell} \in \mathbb{R}^{N_{\ell} \times N_{\ell-1}}$ 为权重矩阵，$b_{\ell} \in \mathbb{R}^{N_{\ell}}$ 为偏置向量，$N_{\ell}$ 表示该层的神经元个数（其中 $N_0 = d$ 为输入维数， $N_L=1$ 为输出维数）。

函数 $\sigma_{\ell}$ 为逐分量激活函数。在本文中主要用到的激活函数包括：
\begin{itemize}
  \item 线性整流单元（Rectified Linear Unit, ReLU），定义为：
  $$\sigma(x) = \mathrm{ReLU}(x) = \max\{0, x\},$$
  \item $k$ 次幂次整流单元（Rectified Power Unit, RePU）定义为 
  $$\sigma(x) = \mathrm{ReLU}^{k}(x) = \max\{0, x\}^k,$$
  其中 $k \in \mathbb{N}^{+}$。
  \item Tanh函数：定义为 $\sigma(x) = \tanh(x) = \frac{e^{x} - e^{-x}}{e^{x} + e^{-x}}$。
\end{itemize}
\end{definition}

\begin{definition}[神经网络的参数]
网络的深度 $\mathcal{D}$ 和宽度 $\mathcal{W}$ 定义为：
\begin{equation*}
\mathcal{D} = L, \quad \mathcal{W} = \max_{\ell=1, \dots, L} N_{\ell}.
\end{equation*}

此外，$\mathcal{B}$ 表示神经网络函数和其导数的公共上界，即
\begin{equation*}
\mathcal{B}=\max_{x\in \Omega}\{|u(x)|,|\nabla u(x)|\}. 
\end{equation*}

网络中所有可训练自由参数的集合记为 $\phi = \{A_{\ell}, b_{\ell}\}_{\ell=1}^{L}$。
\end{definition}

为了刻画逼近理论所需的函数空间，我们进一步定义一类满足特定复杂度约束的神经网络函数集合 $\mathcal{N}(\mathcal{D}, \mathcal{W}, \{k_{i}\}, m)$：

\begin{definition}[神经网络函数类 $\mathcal{N}$]
令 $\mathcal{N}(\mathcal{D}, \mathcal{W}, \{k_{i}\}, m)$ 表示满足以下条件的神经网络函数集合：
\begin{enumerate}
    \item {\heiti 规模约束}：网络实际深度 $D$ 和宽度 $W$ 满足 $D < \mathcal{D}$ 且 $W < \mathcal{W}$（其中 $D, W \in \mathbb{N}^+$）；
    \item {\heiti 激活函数}：所有激活函数均属于族类 $\{\mathrm{ReLU}^{k} \mid k \in \{k_{i}\}\}$；
    \item {\heiti 复杂度下界}：网络的结构参数满足如下不等式约束：
    \begin{equation}
        (D - 2 - \log_{2} W) W \geq m - 1.
    \end{equation}
\end{enumerate}
在上下文明确且不引起混淆的情况下，我们将该函数类简记为 $\mathcal{N}$。
\end{definition}

尽管全连接神经网络具有普适逼近性质，但在处理深层结构时，往往面临梯度消失或梯度爆炸的问题，导致训练困难。为了克服这一挑战，本文在实际计算中采用了残差神经网络（Residual Neural Network, ResNet）。

与方程 \eqref{eq:nn:fcnetwork} 定义的标准前馈结构不同，残差网络引入了“跳跃连接”，使得网络学习的是恒等映射的残差。

\begin{definition}[残差神经网络]
一个包含 $L$ 个残差块的深度残差网络 $v_{\phi}: \mathbb{R}^{d} \to \mathbb{R}$ 定义如下：

首先，通过一个线性变换将输入 $x$ 映射到隐藏层空间：
\begin{equation}
v_{0}(x) = A_{in} x + b_{in},
\end{equation}
其中 $A_{in} \in \mathbb{R}^{m \times d}, b_{in} \in \mathbb{R}^{m}$，此处 $m$ 为残差块的宽度。

随后，数据流经 $L$ 个残差块。第 $\ell$ 个残差块（$\ell=1, \dots, L$）的更新规则为：
\begin{equation}\label{eq:resnet:block}
v_{\ell}(x) = v_{\ell-1}(x) + \mathcal{F}_{\ell}(v_{\ell-1}(x)),
\end{equation}
其中 $\mathcal{F}_{\ell}$ 为残差函数。在最简单的形式中，$\mathcal{F}_{\ell}$ 由单层线性变换与激活函数构成：
\begin{equation}
\mathcal{F}_{\ell}(z) = \sigma\left(A_{\ell} z + b_{\ell}\right),
\end{equation}
这里要求 $A_{\ell} \in \mathbb{R}^{m \times m}, b_{\ell} \in \mathbb{R}^{m}$ 以保证维度匹配，从而实现逐元素相加。

最后，通过输出层将高维特征映射回目标空间：
\begin{equation}
v_{\phi}(x) = A_{out} v_{L}(x) + b_{out},
\end{equation}
其中 $A_{out} \in \mathbb{R}^{1 \times m}, b_{out} \in \mathbb{R}$。
\end{definition}

式 \eqref{eq:resnet:block} 中的 $v_{\ell-1}(x)$ 项通常被称为跳跃连接。这一架构设计的优势在于，它为反向传播过程中的梯度提供了一条无损流通的”短路”路径，缓解了深层网络中常见的梯度消失问题，从而提升了模型的训练稳定性与拟合复杂函数的表达能力。

在后续的数值实验中，我们主要采用基于 $\mathrm{ReLU}^{2}$ 激活函数的残差神经网络来最小化损失函数。现有文献与实证结果均表明，在求解复杂的偏微分方程时，该架构往往表现出优于标准 FNN 的精度与收敛速度。然而，需要指出的是，本文前述的部分理论结果（例如定理 \ref{th4.3} 中的误差估计）是在 FNN 的假设框架下推导得出的。这就引出了一个自然的问题：针对 FNN 建立的理论结果是否能够推广至实验中所使用的 ResNet 架构？

只要注意到恒等映射可以被特定激活函数的线性组合精确表示，ResNet 中的任何一个残差块就可以被等价地重构为具有相同深度的 FNN 结构。具体而言，对于 $\mathrm{ReLU}$ 和 $\mathrm{ReLU}^{2}$ 激活函数，我们拥有以下代数恒等式：
\begin{align}
x &= \mathrm{ReLU}(x) - \mathrm{ReLU}(-x), \\
x &= \frac{1}{2}\left[\mathrm{ReLU}^{2}(x+1) + \mathrm{ReLU}^{2}(-x-1) - \mathrm{ReLU}^{2}(x) - \mathrm{ReLU}^{2}(-x) - 1\right]. \label{eq:relu2_identity}
\end{align}
利用上述恒等式，式 \eqref{eq:resnet:block} 中的恒等路径 $v_{\ell-1}(x)$ 可以被转化为隐藏层的激活输出与线性偏置的组合（如式 \eqref{eq:relu2_identity} 中的常数项 $-1/2$ 可被直接吸收到下一层的偏置向量 $b$ 中）。因此，通过适当扩大网络每一层的宽度，ResNet 的前向传播过程可以等价于一个 FNN。基于这一等价性，前文针对 FNN 所建立的逼近与误差分析理论，只需经过代数推导与常数放缩，即可推广至 ResNet。

\subsection{范数与内积}
接下来，我们介绍常用的范数符号。
\begin{definition}[范数与内积]
设 $\Omega \subset \mathbb{R}^d$ 为有界区域。对于定义在 $\Omega$ 上的函数 $u\in L^2(\Omega)$，其 $L^2$-范数定义为：
\begin{equation*}
\|u\|_{L^2(\Omega)} = \left( \int_{\Omega} |u(x)|^2 \text{dx} \right)^{1/2}.
\end{equation*}
为简便起见，若无特殊说明，下文将 $L^2$-范数简记为 $\|\cdot\|_{L^2}$ 或 $\|\cdot\|$。

对于任意正整数 $m \in \mathbb{N}^+$ 与 $1 \le p < \infty$，Sobolev 空间 $W^{m,p}(\Omega)$ 由所有阶数不超过 $m$ 的弱导数均属于 $L^p(\Omega)$ 的函数组成。借助多重指标 $\alpha$，其 $W^{m,p}$-范数定义为：
$$\|u\|_{W^{m,p}(\Omega)} = \left( \sum_{|\alpha| \le m} \|D^\alpha u\|_{L^p(\Omega)}^p \right)^{1/p}.$$
作为 $W^{m,p}$ 范数的特例，当 $p=2$ 时，$W^{m,2}(\Omega)$ 构成一个 Hilbert 空间，习惯上将其记为 $H^m(\Omega)$。
\begin{equation*}
\|u\|_{H^2(\Omega)}^2 = \|u\|_{L^2(\Omega)}^2 + \sum_{k=1}^d \|\partial_k u\|_{L^2(\Omega)}^2 + \sum_{k=1}^d \sum_{\ell=1}^d \|\partial_k \partial_\ell u\|_{L^2(\Omega)}^2.
\end{equation*}

针对分数阶 Sobolev 空间，函数 $u \in H^{1/2}(\Omega)$ 的范数由其 $L^2$-范数与 Gagliardo 半范数共同构成：
\begin{equation*}
\|u\|_{H^{1/2}(\Omega)}^2 = \|u\|_{L^2(\Omega)}^2 + |u|_{H^{1/2}(\Omega)}^2,
\end{equation*}
其中，衡量半阶平滑度的 Gagliardo 半范数定义为：
\begin{equation*}
|u|_{H^{1/2}(\Omega)}^2 = \iint_{\Omega \times \Omega} \frac{|u(x) - u(y)|^2}{|x - y|^{d+1}} \text{dxdy}.
\end{equation*}

上述空间皆为 Hilbert 空间，其范数均由相应的内积诱导。对应的内积分别记作 $\langle \cdot, \cdot \rangle_{L^2(\Omega)}$、$\langle \cdot, \cdot \rangle_{H^m(\Omega)}$ 与 $\langle \cdot, \cdot \rangle_{H^{1/2}(\Omega)}$，具体定义如下：
\begin{align*}
\langle u, v \rangle_{L^2(\Omega)} &= \int_{\Omega} u(x)v(x) \text{dx}, \\
\langle u, v \rangle_{H^m(\Omega)} &= \sum_{|\alpha| \le m} \langle D^\alpha u, D^\alpha v \rangle_{L^2(\Omega)}, \\
\langle u, v \rangle_{H^{1/2}(\Omega)} &= \langle u, v \rangle_{L^2(\Omega)} + \iint_{\Omega \times \Omega} \frac{(u(x) - u(y))(v(x) - v(y))}{|x - y|^{d+1}}  \text{dxdy}.
\end{align*}

若无混淆，在特定的上下文语境中，内积的下标通常会被省略，统一记作 $\langle \cdot, \cdot \rangle$。
\end{definition}

\begin{definition}[$L^{2}$ 误差]
在数值实验中，为了衡量数值解 $\hat{u}$ 的精度，我们采用如下定义的相对 $L^{2}$ 误差：
\begin{equation*}
\quad\err(\hat{u})=\frac{\|\hat{u}-u^{\dagger}\|_{L^{2}(\Omega)}}{\|u^{\dagger}\|_{L^{2}(\Omega)}},
\end{equation*}
这里，$L^2$ 范数是基于网格点上的值进行近似的，即 $\|v\|_{L^{2}(\Omega)}\approx\sqrt{\frac{1}{N}\sum_{i=1}^{N}v(x_{i})^{2}}$，其中 $\{x_{i}\}_{i=1}^{N}$ 为网格点。
\end{definition}

\subsection{统计机器学习相关定义}
给定来自均匀分布的 $n$ 个独立同分布 (i.i.d.) 样本 $\mathbf{Z}_{n}=\left\{Z_{i}\right\}_{i=1}^{n}$，我们可以利用 Rademacher 复杂度来评估给定函数类 $\mathcal{N}$ 在 $n$ 个随机样本 $\mathbf{Z}_{n}$ 上的限制容量。
\begin{definition}[Rademacher复杂度，Rademacher Complexity]\label{def:staerr:radcom}
设集合 $A \subseteq \mathbb{R}^{n}$，该{\heiti 集合}的经验 Rademacher 复杂度定义为：
$$\mathfrak{R}(A)=\mathbb{E}_{\mathbf{\sigma}}\left[\sup _{a \in A} \frac{1}{n}\left|\sum_{i=1}^{n} \sigma_{i} a_{i}\right|\right]$$
其中 $\{\sigma_{i}\}_{i=1}^{n}$ 是 $n$ 个独立同分布的拉德马赫随机变量，满足 $\mathbb{P}(\sigma_{i}=1)=\mathbb{P}(\sigma_{i}=-1)=1/2$。

进一步地，给定包含 $n$ 个独立同分布样本的随机向量 $\mathbf{Z}_{n}=(Z_1, \dots, Z_n)$，{\heiti 函数类} $\mathcal{N}$ 关于该样本的经验 Rademacher 复杂度定义为：
$$\mathfrak{R}(\mathcal{N})=\mathbb{E}_{\mathbf{Z}_{n}, \mathbf{\sigma}}\left[\sup _{u \in \mathcal{N}} \frac{1}{n} \left| \sum_{i=1}^{n} \sigma_{i} u(Z_{i}) \right|\right].$$
\end{definition}

\begin{definition}[覆盖与覆盖数，Covering Number]\label{def:staerr:cover}
假设 $U \subset \mathbb{R}^{n}$。对于任意 $\varepsilon > 0$，称集合 $V \subset \mathbb{R}^{n}$ 为 $U$ 关于无穷范数距离 $d_{\infty}$ 的一个 $\varepsilon$-覆盖（$\varepsilon$-cover），若对于任意 $u \in U$，均存在 $v \in V$ 使得 $d_{\infty}(u, v)<\varepsilon$，其中距离 $d_{\infty}$ 定义为：
$$d_{\infty}(u, v):=\|u-v\|_{\infty}.$$
集合 $U$ 关于距离 $d_{\infty}$ 的覆盖数记为 $\mathcal{C}(\varepsilon, U, d_{\infty})$，定义为所有可能的 $\varepsilon$-覆盖中基数最小的覆盖所包含的元素个数。
\end{definition}

\begin{definition}[一致覆盖数，Uniform Covering Number]\label{def:staerr:unicovnum}
设 $\mathcal{N}$ 为从输入空间 $\Omega$ 映射到 $\mathbb{R}$ 的函数类。给定 $n$ 个样本 $\mathbf{Z}_{n}=\left(Z_{1}, Z_{2}, \cdots, Z_{n}\right) \in \Omega^{n}$，定义函数类 $\mathcal{N}$ 在样本集 $\mathbf{Z}_{n}$ 上的投影（或经验迹） $\left.\mathcal{N}\right|_{\mathbf{Z}_{n}} \subset \mathbb{R}^{n}$ 为：
$$\left.\mathcal{N}\right|_{\mathbf{Z}_{n}}=\left\{\left(u(Z_{1}), u(Z_{2}), \cdots, u(Z_{n})\right): u \in \mathcal{N}\right\}.$$
函数类 $\mathcal{N}$ 的一致覆盖数 $\mathcal{C}_{\infty}(\varepsilon, \mathcal{N}, n)$ 定义为投影集合在最坏样本分布下的最大覆盖数：
$$\mathcal{C}_{\infty}(\varepsilon, \mathcal{N}, n)=\max_{\mathbf{Z}_{n} \in \Omega^{n}} \mathcal{C}\left(\varepsilon, \left.\mathcal{N}\right|_{\mathbf{Z}_{n}}, d_{\infty}\right).$$
\end{definition}

\begin{definition}[打散，Shattering]\label{def:staerr:shatter}
设 $\mathcal{N}$ 为定义在 $X=\Omega \times \partial\Omega$ 上且取值于 $\{0,1\}$ 的函数族。给定样本集 $S=\left\{x_{1}, x_{2}, \cdots, x_{n}\right\} \subset X$，若对于任意的标签分配 $b \in\{0,1\}^{n}$，均存在函数 $u \in \mathcal{N}$ 满足：
\begin{equation*}
u\left(x_{i}\right)=b_{i}, \quad i=1,2, \ldots, n.
\end{equation*}
则称集合 $S$ 被函数族 $\mathcal{N}$ 打散。
\end{definition}

\begin{definition}[VC 维，VC Dimension]\label{def:staerr:vc_dim}
函数族 $\mathcal{N}$ 的 VC 维（记为 $\operatorname{VCdim}(\mathcal{N})$）定义为能够被 $\mathcal{N}$ 打散的最大样本集合的基数。
\end{definition}

\begin{definition}[伪打散，Pseudo-shattering]\label{def:staerr:pseshatter}
令 $\mathcal{N}$ 为从空间 $\mathcal{X}$ 到 $\mathbb{R}$ 的函数族。设存在样本子集 $S=\left\{x_{1}, x_{2}, \cdots, x_{n}\right\} \subset \mathcal{X}$。若存在一组实数界限 $y_{1}, \cdots, y_{n} \in \mathbb{R}$，使得对于任意符号分配向量 $b \in \{-1, 1\}^{n}$，都至少存在一个函数 $u \in \mathcal{N}$ 满足：
$$\operatorname{sign}\left(u(x_{i})-y_{i}\right)=b_{i}, \quad \forall i=1,2, \ldots, n,$$
则称集合 $S$ 被函数类 $\mathcal{N}$ 伪打散，且称向量 $\mathbf{y} = (y_{1}, \cdots, y_{n})$ 是该打散的见证。
%TODO: 见证这个词不太好，后续可以考虑替换为witness
\end{definition}

\begin{definition}[伪维数，Pseudo-dimension]\label{def:staerr:pdim}
$\mathcal{N}$ 的伪维数（pseudo-dimension）记为 $\operatorname{Pdim}(\mathcal{N})$，定义为被 $\mathcal{N}$ 伪打散的所有集合中的最大基数。
\end{definition}

\subsection{其他定义}
\begin{enumerate}
\item \textbf{渐近记号}：在不等式估计中，使用符号 $A \lesssim B$ 表示存在一个与关键参数无关的有界常数 $C>0$，使得 $A \leq CB$。若该常数依赖于某些特定参数，我们将在文中特别注明。
\item \textbf{大 $\mathcal{O}$ 记号}：若 $A \lesssim B$，则等价记为 $A = \mathcal{O}(B)$。若在此基础上隐藏了对数项，即存在常数 $k > 0$ 使得 $A \lesssim B \cdot \log^k B$，则使用带波浪号的渐近符号记为 $A = \widetilde{\mathcal{O}}(B)$。

\item \textbf{辅助函数}：定义 $\varphi(x) = x\log(x)$ ($x>1$)，并记 $\psi(x)$ 为 $\varphi(x)$ 的反函数。
\end{enumerate}
\begin{remark}
对于上述定义的函数 $\psi(x)$，易证对于任意 $0<k<1$，成立 $\psi(kx) \geq k\psi(x)$。此外，该函数满足如下渐近性质：
$$ \forall \alpha>0, \quad \varphi(x) \lesssim x^{1+\alpha}, \quad \psi(x) \gtrsim x^{1-\alpha}. $$
\end{remark}
